14  package de.funfried.netbeans.plugins.editor.closeleftright.actions.vcs;
15  
16  import javax.swing.AbstractAction;
17  
18  import org.openide.util.NbBundle;
19  import org.openide.util.NbBundle.Messages;
20  import org.openide.util.lookup.ServiceProvider;
21  import org.openide.windows.TopComponent;
22  
23  import de.funfried.netbeans.plugins.editor.closeleftright.AbstractActionFactory;
24  import de.funfried.netbeans.plugins.editor.closeleftright.AdditionalCloseActionFactory;
25  
26  /**
27   * {@link CloseCommitedActionFactory} to create {@link CloseCommitedAction}.
28   *
29   * @author bahlef
30   */
31  @Messages("CTL_CloseCommitedAction=Close All Commited")
32  @ServiceProvider(service = AdditionalCloseActionFactory.class, position = 500)
33  public class CloseCommitedActionFactory extends AbstractActionFactory {
34  	/**
35  	 * Creates a new instance of {@link CloseCommitedActionFactory}.
36  	 */
37  	public CloseCommitedActionFactory() {
38  		super(NbBundle.getMessage(CloseCommitedActionFactory.class, "CTL_CloseCommitedAction"));
39  	}
40  
41  	@Override
42  	public String getId() {
43  		return "closeCommitedAction";
44  	}
45  
46  	@Override
47  	public boolean isGlobalAction() {
48  		return false;
49  	}
50  
51  	@Override
52  	public AbstractAction createAction(TopComponent topComponent) {
53  		return new CloseCommitedAction(getName(), topComponent);
54  	}
55  }
