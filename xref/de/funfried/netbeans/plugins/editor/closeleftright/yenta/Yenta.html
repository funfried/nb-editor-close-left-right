43  package de.funfried.netbeans.plugins.editor.closeleftright.yenta;
44  
45  import java.lang.reflect.Field;
46  import java.lang.reflect.Method;
47  import java.util.Collections;
48  import java.util.HashSet;
49  import java.util.Set;
50  import org.openide.modules.Dependency;
51  import org.openide.modules.ModuleInfo;
52  import org.openide.modules.ModuleInstall;
53  import org.openide.modules.Modules;
54  
55  /**
56   * Fixes up module system dependencies.
57   * <p>First add dependencies on whatever API modules you want to use, even if you are
58   * not (yet) their “friend”. Setting {@code <verifyRuntime>warn</verifyRuntime>} allows you
59   * to still compile against them and declare a regular specification version dependency,
60   * but be careful since this also disables checks about transitive dependency usage.
61   * <p>You can also add dependencies on modules with no exported packages; declare a
62   * specification version dependency and use the same flag to suppress checks.
63   * The same applies to modules exporting friend packages from which you also/instead
64   * want to use implementation packages. In either case be very
65   * careful and preferably catch {@link LinkageError} to be defensive.
66   * <p>Then add a module installer and extend this class rather than {@link ModuleInstall} directly.
67   * Override {@link #friends} and/or {@link #siblings}.
68   * When {@link #validate} is called, module system errors will be suppressed.
69   */
70  public abstract class Yenta extends ModuleInstall {
71  
72      /** Constructor for subclasses. */
73      protected Yenta() {}
74  
75      /**
76       * Specifies the modules with whom you would like to be friends.
77       * These modules must be among your declared dependencies and they must export friend packages.
78       * For each such module, if you are not already a friend, you will be treated as one,
79       * so you will be able to access friend (but not private) packages.
80       * @return a set of module code name bases (default implementation is empty)
81       */
82      protected Set<String> friends() {
83          return Collections.emptySet();
84      }
85  
86      /**
87       * Specifies the modules from whom you need complete access.
88       * These modules must be among your declared dependencies.
89       * For each such module, you will be able to access all packages, as with an implementation dependency.
90       * Be careful to defend against unexpected signature changes!
91       * @return a set of module code name bases (default implementation is empty)
92       */
93      protected Set<String> siblings() {
94          return Collections.emptySet();
95      }
96  
97      /**
98       * @inheritDoc
99       * @throws IllegalStateException if {@link #friends} and {@link #siblings} are misconfigured or if the module system cannot be manipulated
100      */
101     @Override public void validate() throws IllegalStateException {
102         Set<String> friends = friends();
103         Set<String> siblings = siblings();
104         if (friends.isEmpty() && siblings.isEmpty()) {
105             throw new IllegalStateException("Must specify some friends and/or siblings");
106         }
107         ModuleInfo me = Modules.getDefault().ownerOf(getClass());
108         if (me == null) {
109             throw new IllegalStateException("No apparent module owning " + getClass());
110         }
111         try {
112             Object manager = me.getClass().getMethod("getManager").invoke(me);
113             for (String m : friends) {
114                 if (siblings.contains(m)) {
115                     throw new IllegalStateException("Cannot specify the same module " + m + " in both friends and siblings");
116                 }
117                 Object data = data(findDependency(manager, m));
118                 Field friendNamesF = Class.forName("org.netbeans.ModuleData", true, data.getClass().getClassLoader()).getDeclaredField("friendNames");
119                 friendNamesF.setAccessible(true);
120                 Set<?> names = (Set<?>) friendNamesF.get(data);
121                 Set<Object> newNames = new HashSet<Object>(names);
122                 newNames.add(me.getCodeNameBase());
123                 friendNamesF.set(data, newNames);
124             }
125             for (String m : siblings) {
126                 ModuleInfo dep = findDependency(manager, m);
127                 String implVersion = dep.getImplementationVersion();
128                 if (implVersion == null) {
129                     throw new IllegalStateException("No implementation version found in " + m);
130                 }
131                 Object data = data(me);
132                 Field dependenciesF = Class.forName("org.netbeans.ModuleData", true, data.getClass().getClassLoader()).getDeclaredField("dependencies");
133                 dependenciesF.setAccessible(true);
134                 Dependency[] dependencies = (Dependency[]) dependenciesF.get(data);
135                 boolean found = false;
136                 for (int i = 0; i < dependencies.length; i++) {
137                     if (dependencies[i].getName().replaceFirst("/.+$", "").equals(m)) {
138                         Set<Dependency> nue = Dependency.create(Dependency.TYPE_MODULE, dependencies[i].getName() + " = " + implVersion);
139                         if (nue.size() != 1) {
140                             throw new IllegalStateException("Could not recreate dependency from " + dependencies[i] + " based on " + implVersion);
141                         }
142                         dependencies[i] = nue.iterator().next();
143                         found = true;
144                     }
145                 }
146                 if (!found) {
147                     throw new IllegalStateException("Did not find dependency on " + m);
148                 }
149                 // StandardModule.classLoaderUp skips adding a parent if the dep seemed to offer us nothing, and this has already been called.
150                 Object[] publicPackages = (Object[]) dep.getClass().getMethod("getPublicPackages").invoke(dep);
151                 if (publicPackages != null && publicPackages.length == 0) {
152                     me.getClassLoader().getClass().getMethod("append", ClassLoader[].class).invoke(me.getClassLoader(), (Object) new ClassLoader[] {dep.getClassLoader()});
153                 }
154             }
155         } catch (IllegalStateException x) {
156             throw x;
157         } catch (Exception x) {
158             throw new IllegalStateException(x);
159         }
160     }
161 
162     private ModuleInfo findDependency(/*ModuleManager*/Object manager, String m) throws Exception {
163         Object dep = manager.getClass().getMethod("get", String.class).invoke(manager, m);
164         if (dep == null) {
165             throw new IllegalStateException("No such dependency " + m);
166         }
167         return (ModuleInfo) dep;
168     }
169 
170     private Object data(ModuleInfo module) throws Exception {
171         Method dataM = Class.forName("org.netbeans.Module", true, module.getClass().getClassLoader()).getDeclaredMethod("data");
172         dataM.setAccessible(true);
173         return dataM.invoke(module);
174     }
